---
title: "Get Appointment"
api: "GET /api/citas/:id"
description: "Retrieve a specific appointment by ID with full details"
---

## Authentication

This endpoint requires authentication and authorization.

**Required roles:** `admin` or `asistente`

**Header:**
```
Authorization: Bearer <token>
```

## Path Parameters

<ParamField path="id" type="number" required>
  The appointment ID
</ParamField>

## Response

Returns a single appointment object with nested client and doctor information.

<ResponseField name="id" type="number">
  Appointment ID
</ResponseField>

<ResponseField name="cliente_id" type="number">
  Public client ID
</ResponseField>

<ResponseField name="medico_id" type="number | null">
  Assigned doctor ID (null if not confirmed)
</ResponseField>

<ResponseField name="fecha_solicitada" type="string">
  Requested date (ISO 8601)
</ResponseField>

<ResponseField name="hora_solicitada" type="string">
  Requested time (ISO 8601)
</ResponseField>

<ResponseField name="fecha_confirmada" type="string | null">
  Confirmed date (ISO 8601)
</ResponseField>

<ResponseField name="hora_confirmada" type="string | null">
  Confirmed time (ISO 8601)
</ResponseField>

<ResponseField name="sintomas" type="string | null">
  Patient symptoms description
</ResponseField>

<ResponseField name="estado" type="string">
  Appointment status: `pendiente`, `confirmada`, `atendida`, or `cancelada`
</ResponseField>

<ResponseField name="created_at" type="string">
  Creation timestamp
</ResponseField>

<ResponseField name="cliente" type="object">
  Public client information
  
  <ResponseField name="id" type="number">
    Client ID
  </ResponseField>
  
  <ResponseField name="nombres" type="string">
    First name(s)
  </ResponseField>
  
  <ResponseField name="apellidos" type="string | null">
    Last name(s)
  </ResponseField>
  
  <ResponseField name="telefono" type="string">
    Phone number
  </ResponseField>
  
  <ResponseField name="email" type="string | null">
    Email address
  </ResponseField>
</ResponseField>

<ResponseField name="medico" type="object | null">
  Assigned doctor information (null if not confirmed)
  
  <ResponseField name="id" type="number">
    Doctor ID
  </ResponseField>
  
  <ResponseField name="persona_id" type="number">
    Person ID
  </ResponseField>
  
  <ResponseField name="especialidad_id" type="number">
    Specialty ID
  </ResponseField>
  
  <ResponseField name="persona" type="object">
    Doctor's personal information
    
    <ResponseField name="nombres" type="string">
      First name(s)
    </ResponseField>
    
    <ResponseField name="apellidos" type="string">
      Last name(s)
    </ResponseField>
  </ResponseField>
  
  <ResponseField name="especialidad" type="object">
    Medical specialty
    
    <ResponseField name="id" type="number">
      Specialty ID
    </ResponseField>
    
    <ResponseField name="nombre" type="string">
      Specialty name
    </ResponseField>
  </ResponseField>
</ResponseField>

## Error Responses

<ResponseField name="401 Unauthorized">
  Missing or invalid authentication token
  
  ```json
  {
    "message": "Token no proporcionado"
  }
  ```
</ResponseField>

<ResponseField name="403 Forbidden">
  User doesn't have required role
  
  ```json
  {
    "message": "Acceso denegado: requiere rol admin o asistente"
  }
  ```
</ResponseField>

<ResponseField name="404 Not Found">
  Appointment not found
  
  ```json
  {
    "message": "Cita no encontrada"
  }
  ```
</ResponseField>

<ResponseField name="500 Internal Server Error">
  Server error while retrieving appointment
  
  ```json
  {
    "message": "Error al obtener cita"
  }
  ```
</ResponseField>
